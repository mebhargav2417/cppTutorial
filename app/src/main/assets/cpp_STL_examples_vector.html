Example of Vector with some algorithms

#include <iostream>
#include <vector>
#include <algorithm>
#include <string>

using namespace std;

int main() {
     vector<string> v;
     v.push_back("C");
     v.push_back("C++");
     v.push_back("Java");
     v.push_back("Android");
     v.push_back("Dot net");
     v.push_back("C++");

     cout<<"Vector string elements are: ";
     for(unsigned int j=0; j<v.size();j++)
          cout<<v[j]<<"\t";
     cout<<endl;

     v[3] = "C#"; //replaces C# value at 3 or 2+1 position value

     cout<<"After replacing 3rd position element, vector list becomes: ";

     vector <string>::iterator i = v.begin (); //Iterating for vector v

     while (i != v.end ()){
          cout << *i << "\t";
          ++i;
     }

     i = find (v.begin (),v.end (), "C++");

     if (i != v.end ()) {
          int nPosition = distance (v.begin (), i);
          cout <<endl<< "Value "<< *i;
          cout << " found in the vector at position: " << nPosition << endl;
     }

     vector<string> dupV;

     copy(v.begin(),v.end(),back_inserter(dupV)); //copying vector v elements to vector dupV

     cout<<"Vector string elements after copying: ";
     for(unsigned int j=0; j<dupV.size();j++)
          cout<<dupV[j]<<"\t";
     cout<<endl;

     cout<<"Vector string printing from back: ";
     while(!dupV.empty()){
          cout<<dupV.back()<<"\t";
          dupV.pop_back();
     }

     v.insert(v.begin()+3,"python");// using insert()
     v.erase(v.begin()+1);// using erase()
     cout<<endl<<"After inseting and erasing elements in vector v are: ";
     for(unsigned int j=0; j<v.size();j++)
          cout<<v[j]<<"\t";
     return 0;
}

Output:

     Vector string elements are: C C++ Java Android Dot net C++
     After replacing 3rd position element, vector list becomes: C C++ Java C# Dot net C++
     Value C++ found in the vector at position: 1
     Vector string elements after copying: C C++ Java C# Dot net C++
     Vector string printing from back: C++ Dot net C# Java C++ C
     After inseting and erasing elements in vector v are: C Java python C# Dot net C++
